95typedef struct __SOUND_DECODERFUNCTIONS__
96{
97 /* This is a block of info about your decoder. See SDL_sound.h. */
98 const Sound_DecoderInfo info;
99
100 /*
101 * This is called during the Sound_Init() function. Use this to
102 * set up any global state that your decoder needs, such as
103 * initializing an external library, etc.
104 *
105 * Return SDL_TRUE if initialization is successful, SDL_FALSE if
106 * there's a fatal error. If this method fails, then this decoder
107 * is flagged as unavailable until SDL_sound() is shut down and
108 * reinitialized, in which case this method will be tried again.
109 *
110 * Note that the decoders quit() method won't be called if this
111 * method fails, so if you can't intialize, you'll have to clean
112 * up the half-initialized state in this method.
113 */
114 SDL_bool (*init)(void);
115
116 /*
117 * This is called during the Sound_Quit() function. Use this to
118 * clean up any global state that your decoder has used during its
119 * lifespan.
120 */
121 void (*quit)(void);
122
123 /*
124 * Returns non-zero if (sample) has a valid fileformat that this
125 * driver can handle. Zero if this driver can NOT handle the data.
126 *
127 * Extension, which may be NULL, is just a hint as to the form of
128 * data that is being passed in. Most decoders should determine if
129 * they can handle the data by the data itself, but others, like
130 * the raw data handler, need this hint to know if they should
131 * accept the data in the first place.
132 *
133 * (sample)'s (opaque) field should be cast to a Sound_SampleInternal
134 * pointer:
135 *
136 * Sound_SampleInternal *internal;
137 * internal = (Sound_SampleInternal *) sample->opaque;
138 *
139 * Certain fields of sample will be filled in for the decoder before
140 * this call, and others should be filled in by the decoder. Some
141 * fields are offlimits, and should NOT be modified. The list:
142 *
143 * in Sound_SampleInternal section:
144 * Sound_Sample *next; (offlimits)
145 * Sound_Sample *prev; (offlimits)
146 * SDL_RWops *rw; (can use, but do NOT close it)
147 * const Sound_DecoderFunctions *funcs; (that's this structure)
148 * SDL_AudioStream stream; (offlimits)
149 * SDL_bool pending_eof; (offlimits)
150 * SDL_bool pending_error; (offlimits)
151 * void *buffer; (offlimits until read() method)
152 * Uint32 buffer_size; (offlimits until read() method)
153 * void *decoder_private; (read and write access)
154 *
155 * in rest of Sound_Sample:
156 * void *opaque; (this was internal section, above)
157 * const Sound_DecoderInfo *decoder; (read only)
158 * Sound_AudioInfo desired; (read only, usually not needed here)
159 * Sound_AudioInfo actual; (please fill this in)
160 * void *buffer; (offlimits)
161 * Uint32 buffer_size; (offlimits)
162 * Sound_SampleFlags flags; (set appropriately)
163 */
164 int (*open)(Sound_Sample *sample, const char *ext);
165
166 /*
167 * Clean up. SDL_sound is done with this sample, so the decoder should
168 * clean up any resources it allocated. Anything that wasn't
169 * explicitly allocated by the decoder should be LEFT ALONE, since
170 * the higher-level SDL_sound layer will clean up its own mess.
171 */
172 void (*close)(Sound_Sample *sample);
173
174 /*
175 * Get more data from (sample). The decoder should get a pointer to
176 * the internal structure...
177 *
178 * Sound_SampleInternal *internal;
179 * internal = (Sound_SampleInternal *) sample->opaque;
180 *
181 * ...and then start decoding. Fill in up to internal->buffer_size
182 * bytes of decoded sound in the space pointed to by
183 * internal->buffer. The encoded data is read in from internal->rw.
184 * Data should be decoded in the format specified during the
185 * decoder's open() method in the sample->actual field. The
186 * conversion to the desired format is done at a higher level.
187 *
188 * The return value is the number of bytes decoded into
189 * internal->buffer, which can be no more than internal->buffer_size,
190 * but can be less. If it is less, you should set a state flag:
191 *
192 * If there's just no more data (end of file, etc), then do:
193 * sample->flags |= SOUND_SAMPLEFLAG_EOF;
194 *
195 * If there's an unrecoverable error, then do:
196 * __Sound_SetError(ERR_EXPLAIN_WHAT_WENT_WRONG);
197 * sample->flags |= SOUND_SAMPLEFLAG_ERROR;
198 *
199 * If there's more data, but you'd have to block for considerable
200 * amounts of time to get at it, or there's a recoverable error,
201 * then do:
202 * __Sound_SetError(ERR_EXPLAIN_WHAT_WENT_WRONG);
203 * sample->flags |= SOUND_SAMPLEFLAG_EAGAIN;
204 *
205 * SDL_sound will not call your read() method for any samples with
206 * SOUND_SAMPLEFLAG_EOF or SOUND_SAMPLEFLAG_ERROR set. The
207 * SOUND_SAMPLEFLAG_EAGAIN flag is reset before each call to this
208 * method.
209 */
210 Uint32 (*read)(Sound_Sample *sample);
211
212 /*
213 * Reset the decoding to the beginning of the stream. Nonzero on
214 * success, zero on failure.
215 *
216 * The purpose of this method is to allow for higher efficiency than
217 * an application could get by just recreating the sample externally;
218 * not only do they not have to reopen the RWops, reallocate buffers,
219 * and potentially pass the data through several rejecting decoders,
220 * but certain decoders will not have to recreate their existing
221 * state (search for metadata, etc) since they already know they
222 * have a valid audio stream with a given set of characteristics.
223 *
224 * The decoder is responsible for calling seek() on the associated
225 * SDL_RWops. A failing call to seek() should be the ONLY reason that
226 * this method should ever fail!
227 */
228 int (*rewind)(Sound_Sample *sample);
229
230 /*
231 * Reposition the decoding to an arbitrary point. Nonzero on
232 * success, zero on failure.
233 *
234 * The purpose of this method is to allow for higher efficiency than
235 * an application could get by just rewinding the sample and
236 * decoding to a given point.
237 *
238 * The decoder is responsible for calling seek() on the associated
239 * SDL_RWops.
240 *
241 * If there is an error, try to recover so that the next read will
242 * continue as if nothing happened.
243 */
244 int (*seek)(Sound_Sample *sample, Uint32 ms);
245} Sound_DecoderFunctions;
